<script>

  console.log(window.location.search); //?username=andy

  //1. 先去掉问号 substr('起始位置',截取几个字符串);

  let params = location.search.substr(1);//params 就是参数.1-省略（最后一个）

  console.log(params);//username=andy

  //2. 利用等号把字符串分割为数组(自动)

  let arr = params.split('=');

  console.log(arr);//arr[0]=username ; arr[1]=andy

  //3. 把数据写入div中

  var div = document.querySelector('div');

  div.innerHTML=arr[1]+'欢迎您';

</script>
